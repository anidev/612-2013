187 bool Command::Run()
188 {
189  if (!m_runWhenDisabled && m_parent == NULL && DriverStation::GetInstance()->IsDisabled())
190  Cancel();
191 
192  if (IsCanceled())
193  return false;
194 
195  if (!m_initialized)
196  {
197  m_initialized = true;
198  StartTiming();
199  _Initialize();
200  Initialize();
201  }
202  _Execute();
203  Execute();
204  return !IsFinished();
205 }
