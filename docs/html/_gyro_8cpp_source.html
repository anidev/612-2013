28 void Gyro::InitGyro()
29 {
30  if (!m_analog->IsAccumulatorChannel())
31  {
32  wpi_setWPIErrorWithContext(ParameterOutOfRange,
33  "moduleNumber and/or channel (must be accumulator channel)");
34  if (m_channelAllocated)
35  {
36  delete m_analog;
37  m_analog = NULL;
38  }
39  return;
40  }
41 
42  m_voltsPerDegreePerSecond = kDefaultVoltsPerDegreePerSecond;
43  m_analog->SetAverageBits(kAverageBits);
44  m_analog->SetOversampleBits(kOversampleBits);
45  float sampleRate = kSamplesPerSecond *
46  (1 << (kAverageBits + kOversampleBits));
47  m_analog->GetModule()->SetSampleRate(sampleRate);
48  Wait(1.0);
49 
50  m_analog->InitAccumulator();
51  Wait(kCalibrationSampleTime);
52 
53  INT64 value;
54  UINT32 count;
55  m_analog->GetAccumulatorOutput(&value, &count);
56 
57  UINT32 center = (UINT32)((float)value / (float)count + .5);
58 
59  m_offset = ((float)value / (float)count) - (float)center;
60 
61  m_analog->SetAccumulatorCenter(center);
62  m_analog->SetAccumulatorDeadband(0);
63  m_analog->ResetAccumulator();
64 
65  nUsageReporting::report(nUsageReporting::kResourceType_Gyro, m_analog->GetChannel(), m_analog->GetModuleNumber() - 1);
66 }
