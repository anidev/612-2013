32 HiTechnicColorSensor::HiTechnicColorSensor(UINT8 moduleNumber)
33  : m_i2c (NULL)
34 {
35  DigitalModule *module = DigitalModule::GetInstance(moduleNumber);
36  m_mode = kActive;

38  if (module)
