15 void DoubleSolenoid::InitSolenoid()
16 {
17  char buf[64];
18  if (!CheckSolenoidModule(m_moduleNumber))
19  {
20  snprintf(buf, 64, "Solenoid Module %d", m_moduleNumber);
21  wpi_setWPIErrorWithContext(ModuleIndexOutOfRange, buf);
22  return;
23  }
24  if (!CheckSolenoidChannel(m_forwardChannel))
25  {
26  snprintf(buf, 64, "Solenoid Channel %d", m_forwardChannel);
27  wpi_setWPIErrorWithContext(ChannelIndexOutOfRange, buf);
28  return;
29  }
30  if (!CheckSolenoidChannel(m_reverseChannel))
31  {
32  snprintf(buf, 64, "Solenoid Channel %d", m_reverseChannel);
33  wpi_setWPIErrorWithContext(ChannelIndexOutOfRange, buf);
34  return;
35  }
36  Resource::CreateResourceObject(&m_allocated, tSolenoid::kNumDO7_0Elements * kSolenoidChannels);
37 
38  snprintf(buf, 64, "Solenoid %d (Module %d)", m_forwardChannel, m_moduleNumber);
39  if (m_allocated->Allocate((m_moduleNumber - 1) * kSolenoidChannels + m_forwardChannel - 1, buf) == ~0ul)
40  {
41  CloneError(m_allocated);
42  return;
43  }
44  snprintf(buf, 64, "Solenoid %d (Module %d)", m_reverseChannel, m_moduleNumber);
45  if (m_allocated->Allocate((m_moduleNumber - 1) * kSolenoidChannels + m_reverseChannel - 1, buf) == ~0ul)
46  {
47  CloneError(m_allocated);
48  return;
49  }
50  m_forwardMask = 1 << (m_forwardChannel - 1);
51  m_reverseMask = 1 << (m_reverseChannel - 1);
52 
53  nUsageReporting::report(nUsageReporting::kResourceType_Solenoid, m_forwardChannel, m_moduleNumber - 1);
54  nUsageReporting::report(nUsageReporting::kResourceType_Solenoid, m_reverseChannel, m_moduleNumber - 1);
55 }

103 void DoubleSolenoid::Set(Value value)
104 {
105  if (StatusIsFatal()) return;
106  UINT8 rawValue = 0x00;
107 
108  switch(value)
109  {
110  case kOff:
111  rawValue = 0x00;
112  break;
113  case kForward:
114  rawValue = m_forwardMask;
115  break;
116  case kReverse:
117  rawValue = m_reverseMask;
118  break;
119  }
120 
121  SolenoidBase::Set(rawValue, m_forwardMask | m_reverseMask);
122 }

129 DoubleSolenoid::Value DoubleSolenoid::Get()
130 {
131  if (StatusIsFatal()) return kOff;
132  UINT8 value = GetAll();
133 
134  if (value & m_forwardMask) return kForward;
135  if (value & m_reverseMask) return kReverse;
136  return kOff;
