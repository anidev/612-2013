412 void Preferences::ReadTaskRun()
413 {
414  Synchronized sync(m_tableLock);
415  semGive(m_fileOpStarted);
416 
417  std::string comment;
418 
419  FILE *file = NULL;
420  file = fopen(kFileName, "r");
421 
422  if (file != NULL)
423  {
424  std::string buffer;
425  while (true)
426  {
427  char value;
428  do
429  {
430  value = fgetc(file);
431  } while (value == ' ' || value == '\t');
432 
433  if (value == '\n' || value == ';')
434  {
435  if (value == '\n')
436  {
437  comment += "\n";
438  }
439  else
440  {
441  buffer.clear();
442  for (; value != '\n' && !feof(file); value = fgetc(file))
443  buffer += value;
444  buffer += '\n';
445  comment += buffer;
446  }
447  }
448  else if (value == '[')
449  {
450  // Find the end of the section and the new line after it and throw it away
451  for (; value != ']' && !feof(file); value = fgetc(file));
452  for (; value != '\n' && !feof(file); value = fgetc(file));
453  }
454  else
455  {
456  buffer.clear();
457  for (; value != '=' && !feof(file); )
458  {
459  buffer += value;
460  do
461  {
462  value = fgetc(file);
463  } while (value == ' ' || value == '\t');
464  }
465  std::string name = buffer;
466  buffer.clear();
467 
468  bool shouldBreak = false;
469 
470  do
471  {
472  value = fgetc(file);
473  } while (value == ' ' || value == '\t');
474 
475  if (value == '"')
476  {
477  for (value = fgetc(file); value != '"' && !feof(file); value = fgetc(file))
478  buffer += value;
479 
480  // Clear the line
481  while (fgetc(file) != '\n' && !feof(file));
482  }
483  else
484  {
485  for (; value != '\n' && !feof(file);)
486  {
487  buffer += value;
488  do
489  {
490  value = fgetc(file);
491  } while (value == ' ' || value == '\t');
492  }
493  if (feof(file))
494  shouldBreak = true;
495  }
496 
497  std::string value = buffer;
498 
499  if (!name.empty() && !value.empty())
500  {
501  m_keys.push_back(name);
502  m_values.insert(std::pair<std::string, std::string>(name, value));
503  //NetworkTable::GetTable(kTableName)->PutString(name, value);
504 
505  if (!comment.empty())
506  {
507  m_comments.insert(std::pair<std::string, std::string>(name, comment));
508  comment.clear();
509  }
510  }
511 
512  if (shouldBreak)
513  break;
514  }
515  }
516  }
517  else
518  {
519  wpi_setWPIErrorWithContext(NoAvailableResources, "Failed to open preferences file.");
520  }
521 
522  if (file != NULL)
523  fclose(file);
524 
525  if (!comment.empty())
526  m_endComment = comment;
527 }

534 {
535  Synchronized sync(m_tableLock);
536  semGive(m_fileOpStarted);
537 
538  FILE *file = NULL;
539  Priv_SetWriteFileAllowed(1);

552
