67 void ClientConnectionAdapter::reconnect() {
68  {
69  Synchronized sync(LOCK);
70  close();//close the existing stream and monitor thread if needed
71  try{
72  IOStream* stream = streamFactory.createStream();
73  if(stream==NULL)
74  return;
75  connection = new NetworkTableConnection(stream, typeManager);
76  readThread = threadManager.newBlockingPeriodicThread(new ConnectionMonitorThread(*this, *connection), "Client Connection Reader Thread");
77  connection->sendClientHello();
78  gotoState(&ClientConnectionState::CONNECTED_TO_SERVER);
79  } catch(IOException& e){
80  close();//make sure to clean everything up if we fail to connect
81  }
82  }
