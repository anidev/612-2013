35 void Ultrasonic::UltrasonicChecker()
36 {
37  Ultrasonic *u = NULL;
38  while (m_automaticEnabled)
39  {
40  if (u == NULL) u = m_firstSensor;
41  if (u == NULL) return;
42  if (u->IsEnabled())
43  u->m_pingChannel->Pulse(kPingTime); // do the ping
44  u = u->m_nextSensor;
45  Wait(0.1); // wait for ping to return
46  }
47 }

56 void Ultrasonic::Initialize()
57 {
58  bool originalMode = m_automaticEnabled;
59  if (m_semaphore == 0) m_semaphore = semBCreate(SEM_Q_PRIORITY, SEM_FULL);
60  SetAutomaticMode(false); // kill task when adding a new sensor
61  semTake(m_semaphore, WAIT_FOREVER); // link this instance on the list
62  {
63  m_nextSensor = m_firstSensor;
64  m_firstSensor = this;
65  }
66  semGive(m_semaphore);
67 
68  m_counter = new Counter(m_echoChannel); // set up counter for this sensor
69  m_counter->SetMaxPeriod(1.0);
70  m_counter->SetSemiPeriodMode(true);
71  m_counter->Reset();
72  m_counter->Start();
73  m_enabled = true; // make it available for round robin scheduling
74  SetAutomaticMode(originalMode);
75 
76  static int instances = 0;
77  instances++;
78  nUsageReporting::report(nUsageReporting::kResourceType_Ultrasonic, instances);
79 }

165 Ultrasonic::~Ultrasonic()
166 {
167  bool wasAutomaticMode = m_automaticEnabled;
168  SetAutomaticMode(false);
169  if (m_allocatedChannels)
170  {
171  delete m_pingChannel;
172  delete m_echoChannel;
173  }
174  wpi_assert(m_firstSensor != NULL);
175 
176  semTake(m_semaphore, WAIT_FOREVER);
177  {
178  if (this == m_firstSensor)
179  {
180  m_firstSensor = m_nextSensor;
181  if (m_firstSensor == NULL)
182  {
183  SetAutomaticMode(false);
184  }
185  }
186  else
187  {
188  wpi_assert(m_firstSensor->m_nextSensor != NULL);
189  for (Ultrasonic *s = m_firstSensor; s != NULL; s = s->m_nextSensor)
190  {
191  if (this == s->m_nextSensor)
192  {
193  s->m_nextSensor = s->m_nextSensor->m_nextSensor;
194  break;
195  }
196  }
197  }
198  }
199  semGive(m_semaphore);
200  if (m_firstSensor != NULL && wasAutomaticMode)
201  SetAutomaticMode(true);
202 }

213 void Ultrasonic::SetAutomaticMode(bool enabling)
214 {
215  if (enabling == m_automaticEnabled)
216  return; // ignore the case of no change
217 
218  m_automaticEnabled = enabling;
219  if (enabling)
220  {
221  // enabling automatic mode.
222  // Clear all the counters so no data is valid
223  for (Ultrasonic *u = m_firstSensor; u != NULL; u = u->m_nextSensor)
224  {
225  u->m_counter->Reset();
226  }
227  // Start round robin task
228  wpi_assert(m_task.Verify() == false); // should be false since was previously disabled
229  m_task.Start();
230  }
231  else
232  {
233  // disabling automatic mode. Wait for background task to stop running.
234  while (m_task.Verify())
235  Wait(0.15); // just a little longer than the ping time for round-robin to stop
236 
237  // clear all the counters (data now invalid) since automatic mode is stopped
238  for (Ultrasonic *u = m_firstSensor; u != NULL; u = u->m_nextSensor)
239  {
240  u->m_counter->Reset();
241  }
242  m_task.Stop();
243  }
244 }
